from typing import List

class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
 
        diff = [0] * (n + 2)
        
    
        for first, last, seats in bookings:
          
            diff[first] += seats
           
            diff[last + 1] -= seats
        
       
        answer = [0] * n
        current = 0
        
        for i in range(1, n + 1):
            current += diff[i]
            answer[i - 1] = current
        
        return answer
