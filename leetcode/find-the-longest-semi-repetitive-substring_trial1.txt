class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        n = len(s)
        if n <= 2:
            return n
        
        max_len = 2 
        
       
        left = 0
        last_same = -1
        
        for right in range(1, n):
            if s[right] == s[right - 1]:
                if last_same != -1:
                    left = last_same + 1
                
                last_same = right - 1
            
            
            max_len = max(max_len, right - left + 1)
        
        return max_len
