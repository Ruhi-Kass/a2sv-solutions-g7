class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = set('aeiou')
        
       
        current = sum(1 for i in range(k) if s[i] in vowels)
        max_count = current
        
       
        for i in range(k, len(s)):
           
            if s[i - k] in vowels:
                current -= 1
           
            if s[i] in vowels:
                current += 1
            max_count = max(max_count, current)
        
        return max_count
