from typing import List

class Solution:
    def subarraysWithKDistinct(self, nums: List[int], k: int) -> int:
        
        def atMostKDistinct(k: int) -> int:
            freq = [0] * (len(nums) + 1)  
            left = 0
            distinct = 0
            result = 0
            
            for right in range(len(nums)):
                if freq[nums[right]] == 0:
                    distinct += 1
                freq[nums[right]] += 1
                
            
                while distinct > k:
                    freq[nums[left]] -= 1
                    if freq[nums[left]] == 0:
                        distinct -= 1
                    left += 1
                
               
                result += (right - left + 1)
            
            return result
        
      
        return atMostKDistinct(k) - atMostKDistinct(k - 1) if k > 0 else 0
